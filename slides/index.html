Funktionen manchmal heißen sie auch Methoden

		

alert("Achtung!");
confirm("Sicher?");
prompt("Was geht?");

	console.log("kann " + etwas);
}
			

	


	

    
Funktionen/Methoden können Ergebnisse liefern

		


function add(a, b){
	return a + b;
}
			

	


	

    
Bedingungen (1)

		
			
if (person.happy == true) {
	alert(person.name + " ist glücklich!");
}
			
		

	


	

    
Bedingungen (2)

		
			
var tag = "Montag";
if (tag === "Montag") {
	person.happy = true;
} else {
	person.happy = false;
}
if (person.happy) { // implizites true
	alert(person.name + " ist glücklich!");
}
			
		

	


	

    
Bedingungen (3)

		
			
person.happy = tag === "Montag";
person.happy = (tag === "Montag"); // besser zu lesen

if (person.happy) { // implizites true
	alert(person.name + " ist glücklich!");
}
			
		

	


	

    
Bedingungen (4) als Operator

		
			
var tag = person.happy ? "Montag" : "scheinbar nicht Montag";

alert("Heute ist " + tag);
			
		

	


	

    
Iterationen/Schleifen (1)

		
			
for(var i=0; i<7; i++) {
	console.log(wochenTage[i]);
}
			
		

	


	

    
Iterationen/Schleifen (2)

		
			
person.hobbies.map( function (h){
  kann(h);
});
person.hobbies.filter( function (h){
  if(h == "lesen") {
  	confirm("... und schreiben?");
  }
});
